# TypeScript Types

Complete type definitions exported by `@signum/sdk`.

## Core Types

### SignumClient

```typescript
interface SignumClientConfig {
  baseUrl: string;
  timeout?: number;
  retries?: number;
  fetch?: typeof fetch;
  debug?: boolean;
  tokenManager?: TokenManager;
}

class SignumClient {
  constructor(config: SignumClientConfig);
  
  auth: AuthModule;
  accounts: AccountsModule;
  solana: SolanaModule;
  kyc: KycModule;
}
```

### SignumOAuth

```typescript
interface SignumOAuthConfig {
  clientId: string;
  redirectUri: string;
  scopes: string[];
  baseUrl?: string;
}

class SignumOAuth {
  constructor(config: SignumOAuthConfig);
  
  getAuthorizationUrl(): { url: string; codeVerifier: string; state: string };
  exchangeCode(code: string, codeVerifier: string): Promise<TokenResponse>;
  getUserInfo(accessToken: string): Promise<UserInfo>;
  refreshTokens(refreshToken: string): Promise<TokenResponse>;
}
```

## Authentication Types

### AuthSession

```typescript
interface AuthSession {
  accessToken: string;
```

### User

```typescript
interface User {
  id: string;
  email: string;
  displayName?: string;

  createdAt: string;
  chainIdentities?: ChainIdentities;
  kyc?: KycStatus;

```

### TokenResponse

```typescript
interface TokenResponse {
  accessToken: string;
  tokenType: 'Bearer';
  expiresIn: number;
  refreshToken?: string;
  idToken?: string;

}
```

### UserInfo

```typescript
interface UserInfo {
```

## Type Guards

```typescript
import {
  isAuthError,
  isValidationError,
  isSignumError,
  isTokenExpired,
} from '@signum/sdk';

// Usage
if (isAuthError(error)) {
  console.log('Auth error:', error.code);
}

if (isTokenExpired(accessToken)) {
  // Refresh token
}
```
