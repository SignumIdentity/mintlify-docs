# Get Accounts

Retrieve a comprehensive view of the user's profile and all linked chain identities.

## Authorization

<ParamField header="Authorization" type="string" required>
  Bearer token from login or OAuth2 flow.
</ParamField>

## Response

<ResponseField name="user" type="object">
  User profile information.
  <Expandable title="properties">
    <ResponseField name="id" type="string">User ID</ResponseField>
    <ResponseField name="email" type="string">Email address</ResponseField>
    <ResponseField name="displayName" type="string">Display name</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="chains" type="object">
  Chain identities by network.
  <Expandable title="properties">
    <ResponseField name="solana" type="object">
      Solana wallets and attestations.
      <Expandable title="properties">
        <ResponseField name="pda" type="object">
          Primary custody PDA wallet.
        </ResponseField>
        <ResponseField name="linked" type="array">
          Externally linked Solana wallets.
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="evm" type="object">
      EVM wallets (embedded and linked).
      <Expandable title="properties">
        <ResponseField name="embedded" type="object">
          Privy-managed embedded wallet.
        </ResponseField>
        <ResponseField name="linked" type="array">
          Externally linked wallets.
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="canton" type="object">
      Canton Network identity.
      <Expandable title="properties">
        <ResponseField name="partyId" type="string">
          Canton party identifier.
        </ResponseField>
        <ResponseField name="participant" type="string">
          Participant node.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="kyc" type="object">
  KYC verification status.
</ResponseField>

<RequestExample>
```bash cURL
curl https://api.signum.id/accounts \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN"
```

```typescript TypeScript
const response = await fetch('https://api.signum.id/accounts', {
  headers: { Authorization: `Bearer ${accessToken}` },
});

const account = await response.json();
console.log(account.chains.solana.pda.address);
console.log(account.chains.evm.embedded.address);
console.log(account.chains.canton.partyId);
```
</RequestExample>
